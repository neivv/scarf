93    /// Tries to replace a block with a jump to another one which starts in middle of current.
94    pub fn merge_overlapping_blocks(&mut self) {
95        if self.nodes.is_empty() {
96            return;
97        }
98        // TODO: Handle instructions inside instructions
99        for i in 0..self.nodes.len() - 1 {
100            let (left, rest) = self.nodes.split_at_mut(i + 1);
101            let &mut (start_addr, ref mut node) = &mut left[i];
102            let mut current_addr = start_addr + 1;
103            'merge_loop: loop {
104                let &mut (next_addr, ref mut next_node) =
105                    match rest.binary_search_by_key(&current_addr, |x| x.0) {
106                    Ok(o) => &mut rest[o],
107                    Err(e) => {
108                        if e >= rest.len() {
109                            break 'merge_loop;
110                        }
111                        &mut rest[e]
112                    }
113                };
114                if next_addr >= node.end_address {
115                    break 'merge_loop;
116                }
117                let first_default = node.out_edges.default_address();
118                let next_default = next_node.out_edges.default_address();
119                if let (Some(def1), Some(def2)) = (first_default, next_default) {
120                    if def1 == def2 {
121                        node.end_address = next_addr;
122                        let node_is_superset = match (&node.out_edges, &next_node.out_edges) {
123                            (&CfgOutEdges::Branch(..), &CfgOutEdges::Single(..)) => true,
124                            _ => false,
125                        };
126                        if node_is_superset {
127                            next_node.out_edges = mem::replace(
128                                &mut node.out_edges,
129                                CfgOutEdges::Single(NodeLink::new(next_addr))
130                            );
131                        } else {
132                            node.out_edges = CfgOutEdges::Single(NodeLink::new(next_addr));
133                        }
134                        break 'merge_loop;
135                    } else {
136                        // Keep the nodes separate
137                    }
138                } else {
139                    match (&mut node.out_edges, &mut next_node.out_edges) {
140                        (out @ &mut CfgOutEdges::None, &mut CfgOutEdges::None) => {
141                            node.end_address = next_addr;
142                            *out = CfgOutEdges::Single(NodeLink::new(next_addr));
143                            break 'merge_loop;
144                        }
145                        (&mut CfgOutEdges::None, _) | (_, &mut CfgOutEdges::None) => {
146                            error!("Logic error: One of {:x}, {:x} has no out edges, \
147                                   other one does", start_addr.0, next_addr.0);
148                        }
149                        _ => (),
150                    }
151                }
152                current_addr = next_addr + 1;
153            }
154        }
155        self.mark_dirty();
156    }

172    /// Can only be called on a graph with all node links being addresses of one of
173    /// the nodes in graph.
174    pub fn calculate_node_indices(&mut self) {
175        if !self.node_indices_dirty {
176            return;
177        }
178        self.node_indices_dirty = false;
179        for index in 0..self.nodes.len() {
180            let (before, node_pair, after) = {
181                let (before, rest) = self.nodes.split_at_mut(index);
182                let (node, after) = rest.split_first_mut().unwrap();
183                (before, node, after)
184            };
185            let address = node_pair.0;
186            let node = &mut node_pair.1;
187            let end_address = node.end_address;
188            let update_node = |link: &mut NodeLink| {
189                if link.is_unknown() {
190                    return;
191                }
192                link.index = if link.address > address {
193                    // Common case
194                    let next = after.get(0)
195                        .unwrap_or_else(|| panic!(
196                            "Broken graph: {:?} -> {:?} linking to invalid node",
197                            end_address,
198                            link.address,
199                        ));
200                    if next.0 == link.address {
201                        index + 1
202                    } else {
203                        index + 1 + after.binary_search_by_key(&link.address, |x| x.0)
204                            .unwrap_or_else(|_| panic!(
205                                "Broken graph: {:?} -> {:?} linking to invalid node",
206                                end_address,
207                                link.address,
208                            ))
209                    }
210                } else if link.address == address {
211                    index
212                } else {
213                    before.binary_search_by_key(&link.address, |x| x.0)
214                        .unwrap_or_else(|_| panic!(
215                            "Broken graph: {:?} -> {:?} linking to invalid node",
216                            end_address,
217                            link.address,
218                        ))
219                } as u32;
220            };
221            match node.out_edges {
222                CfgOutEdges::Single(ref mut n) => update_node(n),
223                CfgOutEdges::Branch(ref mut n, ref mut cond) => {
224                    update_node(n);
225                    update_node(&mut cond.node);
226                }
227                CfgOutEdges::None => (),
228                CfgOutEdges::Switch(ref mut nodes, _) => {
229                    for node in nodes.iter_mut() {
230                        update_node(node);
231                    }
232                }
233            }
234        }
235        self.entry.index = self.nodes.binary_search_by_key(&self.entry.address, |x| x.0)
236            .expect("Broken graph: Entry was deleted?") as u32;
237    }

239    /// Can only be called on a graph with all node links being addresses of one of
240    /// the nodes in graph.
241    pub fn cycles(&mut self) -> Vec<Vec<NodeLink>> {
242        // This currently generates a bunch of duplicates which are removed at the end.
243        // Could be better.
244        #[derive(Copy, Clone, Eq, PartialEq)]
245        enum CheckState {
246            Unchecked,
247            Checking(u16),
248            Checked,
249        }
250
251        self.calculate_node_indices();
252        self.calculate_distances();
253        let mut nodes = vec![CheckState::Unchecked; self.nodes.len()];
254        let mut chain: Vec<usize> = Vec::with_capacity(self.nodes.len());
255        let mut result = Vec::new();
256        let mut result_first_pos = Vec::new();
257        let mut forks = Vec::new();
258        let mut pos = self.entry.index();
259        'main_loop: loop {
260            let rewind = match nodes[pos] {
261                CheckState::Checked => {
262                    // Nothing to find, rewind
263                    true
264                }
265                CheckState::Checking(chain_pos) => {
266                    let chain_pos = chain_pos as usize;
267                    // Loop found in chain[chain_pos..]
268                    let start = chain[chain_pos..].iter().enumerate().min_by_key(|&(_, &x)| {
269                        self.nodes[x].1.distance
270                    }).map(|(i, _)| chain_pos + i).unwrap();
271                    result.push(
272                        chain[start..].iter().chain(chain[chain_pos..start].iter())
273                            .map(|&x| NodeLink {
274                                address: self.nodes[x].0,
275                                index: x as u32,
276                            }).collect()
277                    );
278                    result_first_pos.push(chain_pos);
279                    true
280                }
281                CheckState::Unchecked => {
282                    nodes[pos] = CheckState::Checking(chain.len() as u16);
283                    chain.push(pos);
284                    if chain.len() >= u16::MAX as usize {
285                        break 'main_loop;
286                    }
287                    let out1 = {
288                        let node = &self.nodes[pos].1;
289                        match node.out_edges {
290                            CfgOutEdges::None => None,
291                            CfgOutEdges::Single(ref s) => s.index_if_not_unkown(),
292                            CfgOutEdges::Branch(ref s, ref cond) => {
293                                if !cond.node.is_unknown() {
294                                    forks.push((pos, cond.node.index()));
295                                }
296                                s.index_if_not_unkown()
297                            }
298                            CfgOutEdges::Switch(ref cases, _) => {
299                                for other in cases.iter().skip(1) {
300                                    forks.push((pos, other.index()));
301                                }
302                                cases.first().map(|x| x.index())
303                            }
304                        }
305                    };
306                    match out1 {
307                        Some(out1) => {
308                            pos = out1;
309                            false
310                        }
311                        None => {
312                            // End, rewind chain
313                            true
314                        }
315                    }
316                }
317            };
318            if rewind {
319                let mut earliest_pos = result_first_pos.last().cloned().unwrap_or(!0);
320                let (rewind_until, other_branch) = match forks.pop() {
321                    Some(s) => s,
322                    None => break 'main_loop,
323                };
324                while let Some(rewind) = chain.last().cloned() {
325                    if rewind == rewind_until {
326                        break;
327                    }
328                    chain.pop();
329                    while chain.len() < earliest_pos && !result_first_pos.is_empty() {
330                        result_first_pos.pop();
331                        earliest_pos = result_first_pos.last().cloned().unwrap_or(!0);
332                    }
333                    if chain.len() < earliest_pos {
334                        nodes[rewind] = CheckState::Checked;
335                    } else {
336                        nodes[rewind] = CheckState::Unchecked;
337                    }
338                }
339                pos = other_branch;
340            }
341        }
342        result.sort();
343        result.dedup();
344        result
345    }

347    /// Retreives a structure which can be used to lookup what nodes branch to current
348    /// node.
349    pub fn predecessors(&mut self) -> Predecessors {
350        self.calculate_node_indices();
351        let mut result = Predecessors {
352            lookup: vec![(!0, !0); self.nodes.len()],
353            long_lists: Vec::new(),
354        };
355        for (idx, node) in self.nodes.iter().enumerate().map(|x| (x.0 as u32, &(x.1).1)) {
356            match node.out_edges {
357                CfgOutEdges::None => (),
358                CfgOutEdges::Single(ref link) => result.add_predecessors(idx, &[*link]),
359                CfgOutEdges::Branch(ref default, ref cond) => {
360                    result.add_predecessors(idx, &[*default, cond.node])
361                }
362                CfgOutEdges::Switch(ref links, _) => result.add_predecessors(idx, &links),
363            }
364        }
365
366        result
367    }

369    pub fn immediate_postdominator<'a>(
370        &'a self,
371        node: NodeIndex<'a, 'e, S>,
372    ) -> Option<NodeBorrow<'a, 'e, S>> {
373        // Do one run through one branch, mark the nodes 1.. based on their distance.
374        // Any branches met store the mark they branched off from.
375        // Then run through other branch until mark > 2 is found, keep track of largest
376        // mark > 2 node of each mark's sub branches.
377        // If a branch doesn't reach any marked node, then the branch's original mark and
378        // all marks after it are set !0 "unaccptable"
379        //
380        // Also mark any visited 0-marked nodes 1, and if another branch reaches 1-marked node,
381        // it can just stop that branch.
382        // Because we stop on 1-marked nodes, it is important to go through any queued
383        // branches from lowest mark to highest, as lower marks override higher ones.
384        //
385        // Once there are no branches to check, the postdominator is received by the first
386        // mark, for which any smaller mark won't reach another mark larger than it through
387        // its child branches.
388        assert!(!self.node_indices_dirty);
389
390        struct State<'a, 'e, S: CfgState> {
391            mark_buf: Vec<u32>,
392            forks: VecDeque<(usize, u32)>,
393            mark_indices: Vec<usize>,
394            nodes: &'a [(Rva, CfgNode<'e, S>)],
395        }
396        let mut state = State {
397            mark_buf: vec![0u32; self.nodes.len()],
398            forks: VecDeque::with_capacity(self.nodes.len()),
399            mark_indices: Vec::with_capacity(32),
400            nodes: &self.nodes,
401        };
402
403        // Return Err on quick exit, Ok for continuing to rest of branches
404        fn traverse_first_branch<'a, 'e, S: CfgState>(
405            state: &mut State<'a, 'e, S>,
406            node: NodeIndex<'a, 'e, S>,
407        ) -> Result<Vec<u32>, Option<NodeBorrow<'a, 'e, S>>> {
408            let mut pos = node.0 as usize;
409            let mut mark = 1;
410            loop {
411                while state.mark_buf[pos] != 0 {
412                    // We reached a loop, that's not useful.
413                    // Switch marks that are larger than last fork to "useless"
414                    let (branch, last_good_mark) = match state.forks.pop_back() {
415                        None => return Err(None),
416                        Some(s) => s,
417                    };
418                    while mark > last_good_mark + 1 {
419                        let index = state.mark_indices.pop().unwrap();
420                        state.mark_buf[index] = !0;
421                        mark -= 1;
422                    }
423                    pos = branch;
424                }
425                state.mark_buf[pos] = mark;
426                state.mark_indices.push(pos);
427                let index = match state.nodes[pos].1.out_edges {
428                    CfgOutEdges::None => None,
429                    CfgOutEdges::Single(ref s) => {
430                        let index = s.index_if_not_unkown();
431                        if state.forks.is_empty() {
432                            if let Some(index) = index.or_else(|| state.mark_indices.get(1).cloned()) {
433                                // Uh, ok, the postdominator is just the next node then.
434                                // Or if a branch was rewinded to start.
435                                return Err(Some(NodeBorrow {
436                                    address: state.nodes[index].0,
437                                    node: &state.nodes[index].1,
438                                    index: NodeIndex(index as u32, PhantomData),
439                                }));
440                            }
441                        }
442                        index
443                    }
444                    CfgOutEdges::Branch(ref s, ref cond) => {
445                        if !cond.node.is_unknown() {
446                            state.forks.push_back((cond.node.index(), mark));
447                        }
448                        s.index_if_not_unkown()
449                    }
450                    CfgOutEdges::Switch(ref cases, _) => {
451                        for other in cases.iter().skip(1) {
452                            state.forks.push_back((other.index(), mark));
453                        }
454                        cases.first().map(|x| x.index())
455                    }
456                };
457                mark += 1;
458                pos = match index {
459                    None => {
460                        let mut mark_limits = vec![!0u32; mark as usize - 1];
461                        for (i, mark) in mark_limits[..mark as usize - 2].iter_mut().enumerate() {
462                            *mark = i as u32 + 2;
463                        }
464                        return Ok(mark_limits);
465                    }
466                    Some(s) => s,
467                };
468            }
469        }
470
471        fn traverse_rest<'a, 'e, S: CfgState>(
472            state: &mut State<'a, 'e, S>,
473            mark_limits: &mut [u32],
474        ) -> Option<NodeBorrow<'a, 'e, S>> {
475            loop {
476                let mut current_mark;
477                let mut pos;
478                loop {
479                    match state.forks.pop_front() {
480                        Some((p, m)) => {
481                            pos = p;
482                            current_mark = m;
483                        }
484                        None => {
485                            let mut end_mark = mark_limits[0];
486                            let mut pos = 2;
487                            while pos < end_mark {
488                                if end_mark == !0 {
489                                    return None;
490                                }
491                                if mark_limits[pos as usize - 1] > end_mark {
492                                    end_mark = mark_limits[pos as usize - 1];
493                                }
494                                pos += 1;
495                            }
496                            if end_mark == !0 {
497                                return None;
498                            }
499                            let index = state.mark_indices[end_mark as usize - 1];
500                            return Some(NodeBorrow {
501                                address: state.nodes[index].0,
502                                node: &state.nodes[index].1,
503                                index: NodeIndex(index as u32, PhantomData),
504                            });
505                        }
506                    };
507                    if mark_limits[current_mark as usize - 1] != !0 {
508                        break;
509                    }
510                    // The mark was known to not reach any other mark at one branch,
511                    // so don't bother.
512                }
513                match state.mark_buf[pos] {
514                    0 => {
515                        state.mark_buf[pos] = 1;
516                        let index = match state.nodes[pos].1.out_edges {
517                            CfgOutEdges::None => None,
518                            CfgOutEdges::Single(ref s) => s.index_if_not_unkown(),
519                            CfgOutEdges::Branch(ref s, ref cond) => {
520                                if cond.node.is_unknown() {
521                                    // Effectively an end where we didn't meet any of the marked nodes
522                                    for mark in mark_limits[current_mark as usize - 1..].iter_mut() {
523                                        *mark = !0;
524                                    }
525                                    None
526                                } else {
527                                    let cond_index = cond.node.index();
528                                    if state.mark_buf[cond_index] != 1 {
529                                        state.forks.push_back((cond_index, current_mark));
530                                    }
531                                    s.index_if_not_unkown()
532                                }
533                            }
534                            CfgOutEdges::Switch(ref cases, _) => {
535                                for other in cases.iter().skip(1) {
536                                    if other.index() != 1 {
537                                        state.forks.push_back((other.index(), current_mark));
538                                    }
539                                }
540                                cases.first().map(|x| x.index())
541                            }
542                        };
543                        match index {
544                            // If we reach an end without seeing a > 2-node, there is no postdominator
545                            None => {
546                                for mark in mark_limits[current_mark as usize - 1..].iter_mut() {
547                                    *mark = !0;
548                                }
549                            }
550                            Some(1) => (),
551                            Some(s) => state.forks.push_back((s, current_mark)),
552                        }
553                    }
554                    1 => (),
555                    other => {
556                        if mark_limits[current_mark as usize - 1] < other {
557                            mark_limits[current_mark as usize - 1] = other;
558                        }
559                    }
560                }
561            }
562        }
563
564        let mut mark_limits = match traverse_first_branch(&mut state, node) {
565            Ok(o) => o,
566            Err(e) => return e,
567        };
568        traverse_rest(&mut state, &mut mark_limits)
569    }

571    /// Converts conditions to resolved form (Relative to branch start).
572    ///
573    /// The actual conversion function
574    /// fn x(condition: Operand, start: Rva, end: Rva)
575    /// has to be provided by caller.
576    pub fn resolve_cond_jump_operands<F>(&mut self, mut convert: F)
577    where F: FnMut(Operand<'e>, Rva, Rva) -> Operand<'e>
578    {
579        for &mut (address, ref mut node) in &mut self.nodes {
580            if let CfgOutEdges::Branch(_, ref mut cond) = node.out_edges {
581                cond.condition = convert(cond.condition, address, node.end_address);
582            }
583        }
584    }
585}

587/// Link indices must be correct.
588fn calculate_node_distance<'e, S: CfgState>(
589    nodes: &mut [(Rva, CfgNode<'e, S>)],
590    buf: &mut Vec<usize>,
591    node_index: usize,
592    distance: u32,
593) {
594    let old_dist = nodes[node_index].1.distance;
595    if old_dist == 0 || old_dist > distance {
596        let orig_len = buf.len();
597        {
598            let node = &mut nodes[node_index].1;
599            node.distance = distance;
600            buf.extend(
601                node.out_edges.out_nodes()
602                    .filter(|x| !x.is_unknown())
603                    .map(|x| x.index()));
604        }
605        let end = buf.len();
606        for pos in orig_len..end {
607            let index = buf[pos];
608            calculate_node_distance(nodes, buf, index, distance + 1);
609        }
610        buf.drain(orig_len..);
611    }
612}

638#[derive(Clone)]
639pub struct CfgNode<'e, State: CfgState> {
640    pub out_edges: CfgOutEdges<'e>,
641    // The address is to the first instruction of in edge nodes instead of the jump-here address,

660#[derive(Debug, Clone)]
661pub enum CfgOutEdges<'e> {
662    None,
663    Single(NodeLink),
664    // The operand is unresolved at the state of jump.
665    // Could reanalyse the block and show it resolved (effectively unresolved to start of the
666    // block) when Cfg is publicly available.
667    Branch(NodeLink, OutEdgeCondition<'e>),
668    // The operand should be a direct, resolved index to the table
669    Switch(Vec<NodeLink>, Operand<'e>),
670}
671
672impl<'e> CfgOutEdges<'e> {
673    pub fn default_address(&self) -> Option<Rva> {
674        match *self {
675            CfgOutEdges::None => None,
676            CfgOutEdges::Single(ref x) => Some(x.address),
677            CfgOutEdges::Branch(ref x, _) => Some(x.address),
678            CfgOutEdges::Switch(..) => None,
679        }
680    }
681
682    pub fn out_nodes<'a>(&'a self) -> CfgOutEdgesNodes<'a, 'e> {
683        CfgOutEdgesNodes(self, 0)
684    }
685}
686
687pub struct CfgOutEdgesNodes<'a, 'e>(&'a CfgOutEdges<'e>, usize);
688
689impl<'a, 'e> Iterator for CfgOutEdgesNodes<'a, 'e> {
690    type Item = &'a NodeLink;
691    fn next(&mut self) -> Option<Self::Item> {
692        let result = match *self.0 {
693            CfgOutEdges::None => None,
694            CfgOutEdges::Single(ref s) => match self.1 {
695                0 => Some(s),
696                _ => None,
697            },
698            CfgOutEdges::Branch(ref node, ref cond) => match self.1 {
699                0 => Some(node),
700                1 => Some(&cond.node),
701                _ => None,
702            },
703            CfgOutEdges::Switch(ref cases, _) => cases.get(self.1),
704        };
705        self.1 += 1;
706        result
707    }
708}

716#[derive(Debug, Clone, Copy, Eq)]
717/// The address is considered a stable link to a node, while index is invalid if the parent
718/// graph's node_indices_dirty is true.
719pub struct NodeLink {
720    address: Rva,
721    index: u32,
722}
723
724impl Ord for NodeLink {
725    fn cmp(&self, other: &Self) -> Ordering {
726        let NodeLink {
727            ref address,
728            index: _,
729        } = *self;
730        address.cmp(&other.address)
731    }
732}
733
734impl PartialOrd for NodeLink {
735    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
736        Some(self.cmp(other))
737    }
738}
739
740impl PartialEq for NodeLink {
741    fn eq(&self, other: &Self) -> bool {
742        let NodeLink {
743            ref address,
744            index: _,
745        } = *self;
746        *address == other.address
747    }
748}

750impl NodeLink {
751    pub fn new(address: Rva) -> NodeLink {
752        NodeLink {
753            address,
754            index: !0,
755        }
756    }
757
758    pub fn unknown() -> NodeLink {
759        NodeLink {
760            address: Rva(u32::MAX),
761            index: !0,
762        }
763    }
764
765    pub fn is_unknown(&self) -> bool {
766        *self == NodeLink::unknown()
767    }
768
769    pub fn address(&self) -> Rva {
770        self.address
771    }
772
773    pub fn index(&self) -> usize {
774        self.index as usize
775    }
776
777    pub fn index_if_not_unkown(&self) -> Option<usize> {
778        if self.is_unknown() {
779            None
780        } else {
781            Some(self.index())
782        }
783    }

786pub struct Predecessors {
787    // Node index to predecessors.
788    //   No predecessors => (!0, !0),
789    //   1 => (idx, !0),
790    //   2 => (idx, idx),
791    //   3+ => (!0, idx to long_lists)
792    lookup: Vec<(u32, u32)>,
793    // Storage when there are more than 2 predecessors.
794    // First idx is current sublist length, followed by predecessors of that length,
795    // followed by !0 to signify end of list or another index to next sublist.
796    long_lists: Vec<u32>,
797}
798
799impl Predecessors {
800    fn add_predecessors(
801        &mut self,
802        predecessor: u32,
803        successors: &[NodeLink],
804    ) {
805        for succ in successors {
806            if succ.index == u32::MAX {
807                continue;
808            }
809            let entry = &mut self.lookup[succ.index as usize];
810            if entry.1 == u32::MAX {
811                // 0 or 1 entries, can just add to the current list
812                if entry.0 == u32::MAX {
813                    entry.0 = predecessor;
814                } else {
815                    entry.1 = predecessor;
816                }
817            } else if entry.0 != u32::MAX {
818                // 2 entries, relocating to long_lists
819                let long_list_idx = self.long_lists.len() as u32;
820                self.long_lists.extend([
821                    3,
822                    entry.0,
823                    entry.1,
824                    predecessor,
825                    !0,
826                ].iter().copied());
827                entry.0 = u32::MAX;
828                entry.1 = long_list_idx;
829            } else {
830                // Already a long list
831                // First seek to last sublist index
832                let mut tail_index = entry.1 as usize;
833                let mut next_location;
834                loop {
835                    let sublist_len = self.long_lists[tail_index];
836                    next_location = tail_index + sublist_len as usize + 1;
837                    let next = self.long_lists[next_location];
838                    if next == u32::MAX {
839                        break;
840                    }
841                    tail_index = next as usize;
842                }
843                if next_location == self.long_lists.len() - 1 {
844                    // Can just expand the list in place
845                    self.long_lists[tail_index] += 1;
846                    self.long_lists[next_location] = predecessor;
847                    self.long_lists.push(u32::MAX);
848                } else {
849                    // Start a new sublist
850                    let new_sublist_idx = self.long_lists.len();
851                    self.long_lists[next_location] = new_sublist_idx as u32;
852                    self.long_lists.extend([
853                        1,
854                        predecessor,
855                        !0,
856                    ].iter().copied());
857                }
858            }
859
860        }
861    }
862
863    /// Iterates through predecessors of a single node.
864    /// `cfg` must be the same Cfg that was used to create this struct.
865    pub fn predecessors<'a, 'e, S: CfgState>(
866        &'a self,
867        cfg: &'a Cfg<'e, S>,
868        link: &NodeLink,
869    ) -> NodePredecessors<'a, 'e, S> {
870        let index = link.index as usize;
871        let predecessors = self.lookup[index];
872        let mode = if predecessors.0 == u32::MAX && predecessors.1 != u32::MAX {
873            PredecessorIterMode::Long(predecessors.1, 0)
874        } else {
875            PredecessorIterMode::Short(predecessors.0, predecessors.1)
876        };
877
878        NodePredecessors {
879            cfg,
880            parent: self,
881            mode,
882        }
883    }
884}
885
886pub struct NodePredecessors<'a, 'e, S: CfgState> {
887    cfg: &'a Cfg<'e, S>,
888    parent: &'a Predecessors,
889    mode: PredecessorIterMode,
890}
891
892#[derive(Copy, Clone)]
893enum PredecessorIterMode {
894    Long(u32, u32),
895    Short(u32, u32),
896}
897
898impl<'a, 'e, S: CfgState> Iterator for NodePredecessors<'a, 'e, S> {
899    type Item = NodeLink;
900    fn next(&mut self) -> Option<Self::Item> {
901        let next_index = match self.mode {
902            PredecessorIterMode::Short(ref mut a, ref mut b) => {
903                if *a == u32::MAX {
904                    return None;
905                }
906                let next_index = *a;
907                *a = *b;
908                *b = u32::MAX;
909                next_index
910            }
911            PredecessorIterMode::Long(ref mut sublist_index, ref mut pos) => {
912                if *sublist_index == u32::MAX {
913                    return None;
914                }
915                let sublist_len = self.parent.long_lists[*sublist_index as usize] as usize;
916                if *pos as usize == sublist_len {
917                    *sublist_index =
918                        self.parent.long_lists[*sublist_index as usize + sublist_len + 1];
919                    *pos = 0;
920                }
921                if *sublist_index == u32::MAX {
922                    return None;
923                }
924                *pos += 1;
925                self.parent.long_lists[*sublist_index as usize + *pos as usize]
926            }
927        };
928        let address = self.cfg.nodes[next_index as usize].0;
929        Some(NodeLink {
930            address,
931            index: next_index,
932        })
933    }
934}
