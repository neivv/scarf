1//! Heapsort implementation from Rust's standard library.
2//! Used over slice::sort, as there are several places in this library
3//! where mostly small vectors are being sorted, and slice::sort adds
4//! unnecessary bloat by pulling in multiple different sorting
5//! algorithms, multiplied by the fact that each different sortable
6//! type causes a new function to be generated.
7
8
9pub fn sort_by<T, F>(v: &mut [T], mut is_less: F)
10    where F: FnMut(&T, &T) -> bool
11{
12    // This binary heap respects the invariant `parent >= child`.
13    let mut sift_down = |v: &mut [T], mut node: usize| {
14        loop {
15            // Children of `node`:
16            let left = node.wrapping_mul(2).wrapping_add(1);
17            let right = node.wrapping_mul(2).wrapping_add(2);
18
19            // Choose the greater child.
20            let greater = if right < v.len() && is_less(&v[left], &v[right]) {
21                right
22            } else {
23                left
24            };
25
26            // Stop if the invariant holds at `node`.
27            if greater >= v.len() || !is_less(&v[node], &v[greater]) {
28                break;
29            }
30
31            // Swap `node` with the greater child, move one step down, and continue sifting.
32            v.swap(node, greater);
33            node = greater;
34        }
35    };
36
37
38    // Build the heap in linear time.
39    for i in (0 .. v.len() / 2).rev() {
40        sift_down(v, i);
41    }
42
43    // Pop maximal elements from the heap.
44    for i in (1 .. v.len()).rev() {
45        v.swap(0, i);
46        sift_down(&mut v[..i], 0);
47    }
48}
49
50pub fn sort<T: std::cmp::Ord>(ops: &mut [T]) {
51    sort_by(&mut *ops, |a, b| a < b);
52}
