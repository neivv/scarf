73pub fn find_functions_with_callers_x86(
74    file: &BinaryFile<VirtualAddress32>,
75) -> Vec<FuncCallPair<VirtualAddress32>> {
76    let mut out = Vec::with_capacity(800);
77    for code in file.code_sections() {
78        let data = &code.data;
79        out.extend(
80            data.iter().enumerate()
81                .filter(|&(_, &x)| x == 0xe8 || x == 0xe9)
82                .flat_map(|(idx, _)| {
83                    data.get(idx + 1..).and_then(|mut x| x.read_u32().ok())
84                        .map(|relative| FuncCallPair {
85                            caller: code.virtual_address + idx as u32,
86                            callee: VirtualAddress32(
87                                code.virtual_address.0
88                                    .wrapping_add((idx as u32 + 5).wrapping_add(relative))
89                            ),
90                        })
91                })
92                .filter(|pair| {
93                    pair.callee >= code.virtual_address &&
94                        pair.callee < code.virtual_address + data.len() as u32
95                })
96        );
97    }
98    out.sort_unstable_by_key(|x| (x.callee, x.caller));
99    out
100}

102pub fn find_functions_with_callers_x86_64(
103    file: &BinaryFile<VirtualAddress64>,
104) -> Vec<FuncCallPair<VirtualAddress64>> {
105    let mut out = Vec::with_capacity(800);
106    for code in file.code_sections() {
107        let data = &code.data;
108        let code_va = code.virtual_address;
109        let code_end = code_va + data.len() as u32;
110        // Ignoring long jumps here, unlike in 32-bit.
111        // Should probably ignore them there as well but I don't want to change it right now.
112        for (idx, bytes) in data.windows(5).enumerate() {
113            if bytes[0] != 0xe8 {
114                continue;
115            }
116            let relative = LittleEndian::read_i32(&bytes[1..5]);
117            let dest = VirtualAddress64(code_va.0.wrapping_add(
118                (idx as i64).wrapping_add(5).wrapping_add(relative as i64) as u64
119            ));
120            if dest >= code_va && dest < code_end {
121                out.push(FuncCallPair {
122                    caller: code_va + idx as u32,
123                    callee: dest,
124                });
125            }
126        }
127    }
128    out.sort_unstable_by_key(|x| (x.callee, x.caller));
129    out
130}

184fn function_ranges_from_exception_info_x86_64_opt(
185    file: &BinaryFile<VirtualAddress64>,
186) -> Option<Vec<(u32, u32)>> {
187    let base = file.base();
188    let mut read = crate::BinaryFileWithCachedSection::try_new(file)?;
189    let pe_header = base + read.read_u32(base + 0x3c)?;
190    let exception_offset = read.read_u32(pe_header + 0xa0)?;
191    let exception_len = read.read_u32(pe_header + 0xa4)?;
192    if exception_len == 0 {
193        return Some(Vec::new());
194    }
195    let exceptions = file.slice_from(exception_offset..exception_offset + exception_len).ok()?;
196    let count = exceptions.chunks_exact(0xc)
197        .filter_map(|chunk| {
198            let unwind_info = base + LittleEndian::read_u32(&chunk[8..]);
199            let flags = read.read_u8(unwind_info)? >> 3;
200            // 0x4 = chain handler
201            (flags & 0x4 == 0).then_some(())
202        })
203        .count();
204    let mut result = Vec::with_capacity(count);
205    let mut iter = exceptions.chunks_exact(0xc);
206    let mut current = loop {
207        match iter.next() {
208            Some(chunk) => {
209                let unwind_info = base + LittleEndian::read_u32(&chunk[8..]);
210                let flags = read.read_u8(unwind_info)? >> 3;
211                if flags & 0x4 != 0 {
212                    continue;
213                }
214                let start = LittleEndian::read_u32(&chunk[..]);
215                let end = LittleEndian::read_u32(&chunk[4..]);
216                break (start, end)
217            }
218            None => return Some(result),
219        }
220    };
221    for chunk in exceptions.chunks_exact(0xc) {
222        let start = LittleEndian::read_u32(&chunk[..]);
223        let end = LittleEndian::read_u32(&chunk[4..]);
224        let unwind_info = base + LittleEndian::read_u32(&chunk[8..]);
225        let flags = read.read_u8(unwind_info)? >> 3;
226        if flags & 0x4 != 0 {
227            // Should be continuing same range as previous
228            if current.1 == start {
229                current.1 = end;
230            }
231        } else {
232            result.push(current);
233            current = (start, end);
234        }
235    }
236    result.push(current);
237    Some(result)
238}

240/// Attempts to find functions of a binary, accepting ones that are called/long jumped to
241/// from somewhere.
242/// Returns addresses relative to start of code section.
243pub fn find_functions<'a, E: ExecutionState<'a>>(
244    file: &BinaryFile<E::VirtualAddress>,
245    relocs: &[E::VirtualAddress],
246) -> Vec<E::VirtualAddress> {
247    let confirmed_ranges = E::function_ranges_from_exception_info(file)
248        .unwrap_or_else(|_| Vec::new());
249    let mut called_functions = Vec::with_capacity(confirmed_ranges.len());
250    for section in file.code_sections() {
251        find_functions_from_calls::<E>(
252            &section.data,
253            section.virtual_address,
254            &mut called_functions,
255        );
256    }
257    // Filter out functions that are inside other function's ranges
258    if !confirmed_ranges.is_empty() {
259        let mut confirmed_pos = confirmed_ranges.iter().cloned();
260        let first = confirmed_pos.next().unwrap();
261        let mut current_start = file.base + first.0;
262        let mut current_end = file.base + first.1;
263        called_functions.sort_unstable();
264        called_functions.dedup();
265        called_functions.retain(|&addr| {
266            loop {
267                if addr <= current_start {
268                    // Either equal to current (Valid function with exception info)
269                    // or less than (Likely a (leaf) function which won't ever unwind)
270                    return true;
271                }
272                if addr < current_end {
273                    // In middle of another function, so it's not a real address
274                    return false;
275                }
276                let next = match confirmed_pos.next() {
277                    Some(s) => s,
278                    None => (u32::MAX, u32::MAX),
279                };
280                current_start = file.base + next.0;
281                current_end = file.base + next.1;
282            }
283        })
284    }
285    called_functions.extend(confirmed_ranges.iter().map(|x| file.base + x.0));
286    called_functions.extend(find_funcptrs(file, relocs).iter().map(|x| x.callee));
287    called_functions.sort_unstable();
288    called_functions.dedup();
289    called_functions
290}

304fn find_funcptrs<Va: VaTrait>(file: &BinaryFile<Va>, relocs: &[Va]) -> Vec<FuncPtrPair<Va>> {
305    let mut out = Vec::with_capacity(4096);
306    let code = file.code_section();
307    for sect in file.sections.iter().filter(|sect| {
308        &sect.name[..] == b".data\0\0\0" ||
309            &sect.name[..] == b".rdata\0\0" ||
310            &sect.name[..] == b".text\0\0\0"
311    }) {
312        collect_relocs_pointing_to_code(code, relocs, sect, &mut out);
313    }
314    out

317fn collect_relocs_pointing_to_code<Va: VaTrait>(
318    code: &BinarySection<Va>,
319    relocs: &[Va],
320    sect: &BinarySection<Va>,
321    out: &mut Vec<FuncPtrPair<Va>>,
322) {
323    let start = match relocs.binary_search(&sect.virtual_address) {
324        Ok(o) => o,
325        Err(e) => e,
326    };
327    let end = match relocs.binary_search(&(sect.virtual_address + sect.data.len() as u32)) {
328        Ok(o) => o,
329        Err(e) => e,
330    };
331    let funcs = relocs[start..end].iter()
332        .flat_map(|&addr| {
333            // TODO broken on x86 for addresses right at end of section as it can't be
334            // read as u64
335            let offset = (addr.as_u64() - sect.virtual_address.as_u64()) as usize;
336            (&sect.data[offset..]).read_u64().ok()
337                .map(|x| (addr, Va::from_u64(x)))
338        })
339        .filter(|&(_src_addr, func_addr)| {
340            let code_size = code.data.len() as u32;
341            func_addr >= code.virtual_address && func_addr < code.virtual_address + code_size
342        })
343        .filter(|&(_src_addr, func_addr)| {
344            // Skip relocs pointing to other relocs, as they are either switch table
345            // or vtable refs in code, which aren't executable code.
346            // Ideally find_funcptrs could also detect switch cases and not check them,
347            // but not sure how to discern that from vtables/fnptr arrays.
348            let is_switch_jump = relocs.binary_search(&func_addr).is_ok();
349            !is_switch_jump
350        })
351        .map(|(src_addr, func_addr)| FuncPtrPair {
352            address: src_addr,
353            callee: func_addr,
354        });
355    out.extend(funcs);
356}

373pub fn find_relocs_x86(
374    file: &BinaryFile<VirtualAddress32>,
375) -> Result<Vec<VirtualAddress32>, crate::OutOfBounds> {
376    let pe_header = file.base + file.read_u32(file.base + 0x3c)?;
377    let reloc_offset = file.read_u32(pe_header + 0xa0)?;
378    let reloc_len = file.read_u32(pe_header + 0xa4)?;
379    let relocs = file.slice_from(reloc_offset..reloc_offset + reloc_len)?;
380    let mut result = Vec::new();
381    let mut offset = 0;
382    while offset < relocs.len() {
383        let rva = try_get!(relocs, offset..).read_u32()?;
384        let base = file.base + rva;
385        let size = try_get!(relocs, offset + 4..).read_u32()? as usize;
386        if size < 8 {
387            break;
388        }
389        let block_relocs = try_get!(relocs, offset + 8..offset + size);
390        for mut reloc in block_relocs.chunks_exact(2) {
391            if let Ok(c) = reloc.read_u16() {
392                if c & 0xf000 == 0x3000 {
393                    result.push(base + u32::from(c & 0xfff));
394                }
395            }
396        }
397        offset += size;
398    }
399    result.sort_unstable();
400    Ok(result)
401}

439/// The returned array is sorted by value.
440pub fn relocs_with_values<Va: VaTrait>(
441    file: &BinaryFile<Va>,
442    mut relocs: &[Va],
443) -> Result<Vec<RelocValues<Va>>, crate::OutOfBounds> {
444    let mut result = Vec::with_capacity(relocs.len());
445    'outer: while !relocs.is_empty() {
446        let (section, reloc_count, start_address) = {
447            if let Some(section) = file.section_by_addr(relocs[0]) {
448                let end = section.virtual_address + section.data.len() as u32;
449                let count = relocs.binary_search(&end).unwrap_or_else(|x| x);
450                (&section.data, count, section.virtual_address)
451            } else {
452                relocs = &relocs[1..];
453                continue 'outer;
454            }
455        };
456        let values = (&relocs[..reloc_count]).iter().map(|&address| {
457            let relative = (address.as_u64().wrapping_sub(start_address.as_u64())) as usize;
458            let value = if Va::SIZE == 4 {
459                Va::from_u64(
460                    section.get(relative..relative.wrapping_add(4))
461                        .and_then(|mut x| x.read_u32().ok())
462                        .unwrap_or(0) as u64
463                )
464            } else {
465                Va::from_u64(
466                    section.get(relative..relative.wrapping_add(8))
467                        .and_then(|mut x| x.read_u64().ok())
468                        .unwrap_or(0)
469                )
470            };
471            RelocValues {
472                address,
473                value,
474            }
475        });
476        result.extend(values);
477        relocs = &relocs[reloc_count..];
478    }
479    result.sort_unstable_by_key(|x| x.value);
480    Ok(result)
481}

483/// A trait for additional branch-specific analysis state that will be merged by the
484/// analysis when two different branches join.
485///
486/// See [state merging](../analysis/struct.FuncAnalysis.html#state-merging-and-loops)
487/// for more details on how analysis merges states.
488///
489/// At the moment, only differences in `ExecutionState` will cause analysis to merge
490/// and recheck a branch; there is no way yet for `AnalysisState` to declare two states
491/// different to require a recheck when there hasn't been changes in `ExecutionState`.
492///
493/// You can access and modify the `AnalysisState` for current branch in [`Analyzer`] callbacks
494/// by calling [`Control::user_state`].
495pub trait AnalysisState: Clone {
496    /// Called when analysis merges states of two different branches.
497    ///
498    /// The implementation is expected to update `self` with values that are a merge of
499    /// `self` and `newer`, but the implementation may do anything it wants, as scarf does not
500    /// touch `AnalysisState` at all.
501    fn merge(&mut self, newer: Self);
502}
503
504/// A no-op zero-sized type implementing [`AnalysisState`] that can be used if there
505/// is no need for user-side mergeable state.
506#[derive(Default, Clone, Debug)]
507pub struct DefaultState;
508
509impl AnalysisState for DefaultState {
510    fn merge(&mut self, _newer: Self) {
511    }
512}

514/// The analysis runner of scarf.
515///
516/// `FuncAnalysis` executes through all paths ('branches') of a function, calling
517/// user-defined callbacks in [`Analyzer`] for every state-modifying operation, while
518/// keeping track of what branches have been already executed.
519///
520/// # State merging and loops
521///
522/// When two separate branches of execution converge, `FuncAnalysis` will merge the
523/// [`ExecutionState`] of branches: Any [`Operand`] in registers or in memory that differs
524/// will be replaced with `OperandType::Undefined` if it wasn't `Undefined` already. If this
525/// caused anything in `ExecutionState` to change, the following branch will be analyzed
526/// again, even if analysis had already walked through its instruction once.
527///
528/// Scarf does not have any special understanding of loops, a jump backwards is just another
529/// merge point. This effectively often makes scarf to provide user code an `ExecutionState`
530/// that reports registers to contain constants and jumps to be always/never taken for first
531/// iteration of a loop, and later on run the branch 'correctly' with `Undefined` when the
532/// jump backwards has been seen.
533///
534/// ## Example
535///
536/// The following code is a `strlen` implementation in x86-64 assembly. Register `rcx` has
537/// the null-terminated input string, and `rax` will contain the string length on return.
538///
539/// ```asm
540/// entry:
541///     mov rax, 0
542/// loop:
543///     movzx rdx, byte [rcx]
544///     add rax, 1
545///     add rcx, 1
546///     cmp rdx, 0
547///     jne loop
548/// exit:
549///     sub rax, 1 ; Account for null byte having been counted above
550///     ret
551/// ```
552///
553/// Scarf considers there to 3 branches:
554///
555/// 1) Branch from `entry` to `exit`
556/// 2) Branch from `loop` to `exit`
557/// 3) Branch from `exit` that returns
558///
559/// Note that both branch 1 and 2 include the loop instructions. In this case, the backwards
560/// jump can only be known to be possibly taken after the first branch has already been
561/// simulated, so there would have been no way for scarf to know that `movzx rdx, byte [rcx]`
562/// will be a start of another branch. For consistency, even if a jump to current instruction
563/// has been seen, scarf will walk each branch until it sees a jump or return instruction.
564///
565/// On analysis start, scarf will execute branch 1, seeing that on jump the execution splits
566/// to two new branches. [Resolving](ExecutionState::resolve) `rax` during this will return
567/// constant 0, later on 1, and resolving the jump condition at the end of the branch
568/// gives `(Mem8[rcx] == 0) == 0` - "Jump if first byte of argument `rcx` is nonzero".
569///
570/// After the first branch, both branches 2 and 3 are queued; there is no guarantee
571/// which will be analyzed first. If branch 3 is first, scarf will report the function to
572/// return (resolving `rax`) constant 0. Once branch 2 is analyzed, it will have similar
573/// 'known' results what branch 1 shows (`rax` changes from 1 to 2, jump condition is
574/// `(Mem8[rcx + 1] == 0) == 0`), as this is the first time branch has started from there.
575///
576/// As branch 2 analysis reaches the jump, scarf queues branches 2 and 3, and notices that the
577/// values in registers `rax`, `rcx`, and `rdx` differ from what the older branches 2 and 3 had.
578/// Each of those registers will be assigned an `Undefined` value for the new queued branches.
579/// If branch 3 wasn't analyzed yet, the state with `Undefined` overwrites the earlier queued
580/// 'will-return-zero' state.
581///
582/// Finally, scarf walks through branches 2 and 3 with the undefined state (in either order).
583/// At end of this second run of branch 2, `rax` will have value `Undefined_a + 1`, rcx has
584/// `Undefined_b + 1`, and rdx has `Mem8[Undefined_b]`. These do not add anything over the
585/// states where these registers were `Undefined`, so no more branches will be queued.
586///
587/// Since branch 3 contains `sub rax, 1` instruction, the final return value reported by
588/// scarf is the not-very-helpful `Undefined - 1`.
589///
590/// ## Other merge details
591///
592/// In general, [`Operand`] merging is defined as "If old value is undefined or differs from new,
593/// merge to unique undefined value". However, there are few details that are worth noting:
594///
595/// - Undefined values allocated during `FuncAnalysis::analyze` call will get reused after
596///     the call returns. This is mainly worth noting if user code returns results as `Operand`,
597///     and compares them to values from other analysis runs.
598/// - If old or new value is undefined, the merged value may recycle either of those values, or
599///     allocate a new unique undefined.
600/// - When values in memory are merged, a single unit of `Operand` to be merged is not a byte,
601///     but a word (4 or 8 bytes, depending on the `ExecutionState`). This means that if code
602///     has byte-sized globals or structure fields, they may be assigned `Undefined` because
603///     values neighbour address differed. So far this has rarely ended up causing issues, but
604///     it may sometimes need special handling from user code.
605/// - If the address operand of memory contains `Undefined` at all, the merge operation may
606///     drop the value entirely, reverting it back to 'original'.
607///     - That is, while a write of `5` to `Mem32[Undefined + 4]` is guaranteed to resolve to `5`
608///     in same branch as the write, later branches may resolve just to `Mem32[Undefined + 4]`
609///     again.
610pub struct FuncAnalysis<'a, Exec: ExecutionState<'a>, State: AnalysisState> {
611    binary: &'a BinaryFile<Exec::VirtualAddress>,
612    cfg: Cfg<'a, Exec, State>,
613    unchecked_branches: BTreeMap<Rva, (Exec, State, PrecedingBranchAddr)>,
614    // Branches which are before current address.
615    // The goal is to reduce amount of work when a switch jumps backwards
616    // at end of the case, doing all unchecked_branches before promoting
617    // more_unchecked_branches to new unchecked_branches.
618    // Though this means that some other function layouts end
619    // up doing more work. Not sure if there's a good solution.
620    //
621    // This current solution is a major (I saw over 50% less work) improvement over
622    // the previous solution which always just picked branch with lowest address as next.
623    // Another thing I tried is to enable this only when a function contained a switch
624    // jump; result was slightly worse than now. (In other words, this is a major
625    // improvement for functions with switches, minor for others)
626    //
627    // E.g. this is good for functions looking like
628    //
629    //  switch:
630    //      jmp [switch_table + case * 4]
631    //  case0:
632    //      xx
633    //      jmp switch
634    //  case1:
635    //      xx
636    //      je case1_b:
637    //      yy
638    //  case1_b:
639    //      jmp switch
640    //  etc..
641    // Avoiding reanalyzing from start of the switch after every time a jump reaches it
642    //
643    // But this is worse for a function looking like
644    //
645    //  loop1:
646    //      xx
647    //      dec eax
648    //      jne loop1
649    //      mov eax, 6
650    //      (lot of code)
651    //  loop2:
652    //      xx
653    //      dec eax
654    //      jne loop2
655    // as it'll run to the end once, then starts from loop1 again
656    more_unchecked_branches: BTreeMap<Rva, (Exec, State, PrecedingBranchAddr)>,
657    current_branch: Exec::VirtualAddress,
658    operand_ctx: OperandCtx<'a>,
659    merge_state_cache: MergeStateCache<'a>,
660}

662/// The analysis will assume that if only a single branch has lead to a branch,
663/// then merging of execution states can be skipped and it will just be assumed
664/// that `state = merge(old, new)` can be replaced with just `state = new`, e.g.
665/// only thing that should have changed when the single preceding branch is ran a
666/// second time is that more values are now undefined.
667///
668/// If value of this newtype is u32::MAX, then there are multiple preceding branches
669/// and states to branch using this PrecedingBranchAddr have to be merged.
670/// Otherwise this is an address of the preceding branch relative to binary.base().
671#[derive(Copy, Clone)]
672struct PrecedingBranchAddr(u32);
673
674impl PrecedingBranchAddr {
675    fn multiple() -> PrecedingBranchAddr {
676        PrecedingBranchAddr(u32::MAX)
677    }
678
679    fn is_multiple(self) -> bool {
680        self.0 == u32::MAX
681    }
682
683    fn are_same(self, other: PrecedingBranchAddr) -> bool {
684        self.0 == other.0 && self.0 != u32::MAX
685    }
686}

1038/// The trait that will be implemented by user-side code to hook into [`FuncAnalysis`] to
1039/// retrieve whichever results are wanted.
1040///
1041/// Most commonly the `Analyzer` will implement at least the `operation()` hook,
1042/// which will be called for each sub-step of each instruction `FuncAnalysis` is walking through.
1043///
1044/// During the hooks the `Analyzer` has access to [`Control`], which is used to read and modify
1045/// any scarf-side analysis state available to user code, as well as a `&mut self` reference
1046/// to the `Analyzer` structure, which can contain any state the user code needs to be shared
1047/// across entire analysis. If the user code needs branch-specific state with custom
1048/// [state merging](../analysis/struct.FuncAnalysis.html#state-merging-and-loops) behaviour,
1049/// [`Analyzer::State`] can be used to easily implement the state.
1050pub trait Analyzer<'e> : Sized {
1051    /// The `ExecutionState`, CPU architecture that will be used by this analyzer.
1052    type Exec: ExecutionState<'e>;
1053
1054    /// Additional user-defined state that will be
1055    /// [merged by analysis](../analysis/struct.FuncAnalysis.html#state-merging-and-loops) when
1056    /// two branches of execution join.
1057    ///
1058    /// If not needed, [`DefaultState`] can be used to get a state which carries no information.
1059    ///
1060    /// When not using `DefaultState`, call [`FuncAnalysis::custom_state`] to create the
1061    /// `FuncAnalysis` struct. The state can be read and modified during all Analyzer hooks with
1062    /// [`Control::user_state`].
1063    type State: AnalysisState;
1064
1065    /// Called for each `Operation` of disassembled instruction.
1066    ///
1067    /// This is the main function for user code's analysis inspection and manipulation.
1068    fn operation(&mut self, _control: &mut Control<'e, '_, '_, Self>, _op: &Operation<'e>) {}
1069
1070    /// Called each time the analysis starts executing instructions from a new branch.
1071    fn branch_start(&mut self, _control: &mut Control<'e, '_, '_, Self>) {}
1072
1073    /// Called each time the analysis has finished a branch, either due to `Operation::Jump`
1074    /// or `Operation::Return`.
1075    ///
1076    /// Not called if user has requested the branch to be stopped through [`Control::end_branch`].
1077    fn branch_end(&mut self, _control: &mut Control<'e, '_, '_, Self>) {}
1078}

1132        state: (Exec, State),
1133    ) -> FuncAnalysis<'a, Exec, State> {

1136            cfg: Cfg::new(),

1140                map.insert(Rva(rva), (state.0, state.1, PrecedingBranchAddr::multiple()));
1141                map
1142            },
1143            more_unchecked_branches: BTreeMap::new(),
1144            current_branch: start_address,
1145            operand_ctx: ctx,
1146            merge_state_cache: MergeStateCache::new(),
1147        }

1150    fn pop_next_branch_merge_with_cfg(
1151        &mut self,
1152    ) -> Option<(Rva, (Exec, State), PrecedingBranchAddr)> {
1153        while let Some((rva, mut branch_state, preceding)) = self.pop_next_branch() {
1154            match self.cfg.get_state(rva) {
1155                Some(state) => {
1156                    if state.preceding_branch.are_same(preceding) {
1157                        let mut user_state = state.data.1.clone();
1158                        user_state.merge(branch_state.1);
1159                        return Some((rva, (branch_state.0, user_state), preceding));
1160                    } else {
1161                        state.preceding_branch = PrecedingBranchAddr::multiple();
1162                        state.data.0.maybe_convert_memory_immutable(16);
1163                        let merged = Exec::merge_states(
1164                            &mut state.data.0,
1165                            &mut branch_state.0,
1166                            &mut self.merge_state_cache,
1167                        );
1168                        match merged {
1169                            Some(s) => {
1170                                let mut user_state = state.data.1.clone();
1171                                user_state.merge(branch_state.1);
1172                                return Some((
1173                                    rva,
1174                                    (s, user_state),
1175                                    PrecedingBranchAddr::multiple(),
1176                                ));
1177                            }
1178                            // No change, take another branch
1179                            None => (),
1180                        }
1181                    }
1182                }
1183                None => {
1184                    return Some((rva, branch_state, preceding));
1185                }
1186            }
1187        }
1188        None
1189    }

1204    /// Runs the analysis, calling the user-defined callbacks defined on [`Analyzer`] trait
1205    /// as the code is being stepped thorugh.
1206    pub fn analyze<A: Analyzer<'a, State = State, Exec = Exec>>(&mut self, analyzer: &mut A) {
1207        let old_undef_pos = self.operand_ctx.get_undef_pos();
1208        let mut disasm = Exec::Disassembler::new(
1209            self.operand_ctx,
1210            self.binary,
1211            self.current_branch,
1212        );
1213
1214        while let Some((addr, state, preceding)) =
1215            self.pop_next_branch_and_set_disasm(&mut disasm)
1216        {
1217            self.current_branch = addr;
1218            let end = self.analyze_branch(analyzer, &mut disasm, addr, state, preceding);
1219            if end == Some(End::Function) {
1220                break;
1221            }
1222        }
1223        self.operand_ctx.set_undef_pos(old_undef_pos);
1224    }

1234    /// Disasm must have been set to `addr`
1235    fn analyze_branch<'b, A: Analyzer<'a, State = State, Exec = Exec>>(
1236        &mut self,
1237        analyzer: &mut A,
1238        disasm: &mut Exec::Disassembler,
1239        addr: Exec::VirtualAddress,
1240        state: (Exec, State),
1241        preceding_branch: PrecedingBranchAddr,
1242    ) -> Option<End> {
1243        // update_analysis_for_operation is a small function, which would be cleaner
1244        // to inline here if it keeping it separate wasn't good for binary size
1245        // (It is not generic over A).
1246        // It mainly takes state (ControlInner) by ref, but on branches it will move
1247        // exec state out and leave AnalysisUpdateResult::End there, which this function
1248        // uses as a signal to break out of instruction loop.
1249        let mut inner_wrap = AnalysisUpdateResult::Continue(ControlInner {
1250            analysis: self,
1251            address: addr,
1252            branch_start: addr,
1253            continue_at_address: Rva(0),
1254            instruction_length: 0,
1255            skip_operation: false,
1256            state,
1257            end: None,
1258        });
1259        let mut control = match inner_wrap {
1260            AnalysisUpdateResult::Continue(ref mut inner) => Control {
1261                inner,
1262            },
1263            // Unreachable
1264            AnalysisUpdateResult::End(end) => return end,
1265        };
1266        analyzer.branch_start(&mut control);
1267        if control.inner.end.is_some() {
1268            return control.inner.end;
1269        }
1270
1271        // Create CfgNode in advance to avoid some small moves.
1272        // state doesn't get changed after creation, other fields do.
1273        let mut node = Some(CfgNode {
1274            state: CfgState {
1275                phantom: Default::default(),
1276                data: control.inner.state.clone(),
1277                preceding_branch,
1278            },
1279            out_edges: CfgOutEdges::None,
1280            end_address: Rva(0),
1281            distance: 0,
1282        });
1283        // add_resolved_constraint_from_unresolved is not expected to be useful
1284        // when only one branch leads to this branch, but multiple branches merging
1285        // may end up having same unresolved constraint with different resolved
1286        // constraints.
1287        if preceding_branch.is_multiple() {
1288            control.inner.state.0.add_resolved_constraint_from_unresolved();
1289        }
1290        loop {
1291            let address = disasm.address();
1292            control.inner.address = address;
1293            let instruction = disasm.next();
1294            control.inner.instruction_length = instruction.len() as u8;
1295            for op in instruction.ops() {
1296                control.inner.skip_operation = false;
1297                analyzer.operation(&mut control, op);
1298                if let Some(end) = control.inner.end {
1299                    let end_now = end != End::ContinueBranch || is_branch_end_op(op);
1300                    if end_now {
1301                        update_analysis_for_user_requested_end(&mut inner_wrap, &mut node);
1302                        return Some(end);
1303                    }
1304                }
1305                if control.inner.skip_operation {
1306                    continue;
1307                }
1308                if is_branch_end_op(op) {
1309                    analyzer.branch_end(&mut control);
1310                }
1311                update_analysis_for_operation(&mut inner_wrap, op, &mut node);
1312                control = match inner_wrap {
1313                    AnalysisUpdateResult::End(end) => {
1314                        return end;
1315                    }
1316                    AnalysisUpdateResult::Continue(ref mut inner) => Control {
1317                        inner,
1318                    },
1319                };
1320            }
1321            if let Some(end) = control.inner.end {
1322                update_analysis_for_user_requested_end(&mut inner_wrap, &mut node);
1323                return Some(end);
1324            }
1325        }
1326    }

1328    fn add_unchecked_branch(
1329        &mut self,
1330        rva: Rva,
1331        mut state: (Exec, State),
1332    ) {
1333        use std::collections::btree_map::Entry;
1334        let base = self.binary.base();
1335        let queue = match base + rva.0 < self.current_branch {
1336            true => &mut self.more_unchecked_branches,
1337            false => &mut self.unchecked_branches,
1338        };
1339
1340        let preceding_addr = self.current_branch.as_u64()
1341            .checked_sub(base.as_u64())
1342            .and_then(|x| Some(PrecedingBranchAddr(u32::try_from(x).ok()?)))
1343            .unwrap_or(PrecedingBranchAddr::multiple());
1344        match queue.entry(rva) {
1345            Entry::Vacant(e) => {
1346                e.insert((state.0, state.1, preceding_addr));
1347            }
1348            Entry::Occupied(mut e) => {
1349                let val = e.get_mut();
1350                if preceding_addr.are_same(val.2) {
1351                    val.0 = state.0;
1352                    val.1.merge(state.1);
1353                } else {
1354                    val.2 = PrecedingBranchAddr::multiple();
1355                    let result =
1356                        Exec::merge_states(&mut val.0, &mut state.0, &mut self.merge_state_cache);
1357                    if let Some(new) = result {
1358                        val.0 = new;
1359                        val.1.merge(state.1);
1360                    }
1361                }
1362            }
1363        }
1364    }

1366    fn pop_next_branch(&mut self) ->
1367        Option<(Rva, (Exec, State), PrecedingBranchAddr)>
1368    {
1369        if self.unchecked_branches.is_empty() {
1370            std::mem::swap(&mut self.unchecked_branches, &mut self.more_unchecked_branches);
1371        }
1372        let (addr, state) = self.unchecked_branches.pop_first()?;
1373        Some((addr, (state.0, state.1), state.2))
1374    }

1377    ///
1378    /// Calling this is not necessary if you do not need a `Cfg`.
1379    ///
1380    /// Currently if [`Control::end_analysis`] has been used to stop the analysis,
1381    /// any remaining branches (but not the one that was being analyzed on `end_analysis` call!)
1382    /// will be walked through to gain a 'better' idea of the control flow graph. This is somewhat

1389    /// Runs analyzer to end with any user interaction, producing `Cfg`.
1390    ///
1391    /// Maybe to be removed later? User code should ideally not use `end_analysis` and
1392    /// expect that any pending branches get processed by finish().
1393    pub fn finish_with_changes<F>(
1394        mut self,
1395        mut hook: F
1396    ) -> Cfg<'a, Exec, State>
1397    where F: FnMut(
1398        &Operation<'a>,
1399        &mut Exec,
1400        Exec::VirtualAddress,
1401    ) {
1402        let mut analyzer = RunHookAnalyzer {
1403            phantom: Default::default(),
1404            hook: &mut hook,
1405        };
1406        self.analyze(&mut analyzer);
1407
1408        let mut cfg = self.cfg;
1409        cfg.merge_overlapping_blocks();
1410        let binary = self.binary;
1411        let ctx = self.operand_ctx;
1412        cfg.resolve_cond_jump_operands(|condition, rva, end_rva| {
1413            let address = binary.base() + rva.0;
1414            let end_address = binary.base() + end_rva.0;
1415            let mut analysis = FuncAnalysis::new(binary, ctx, address);
1416            let mut analyzer = FinishAnalyzer {
1417                hook: &mut hook,
1418                end_address,
1419                result: condition.clone(),
1420                condition,
1421            };
1422            analysis.analyze(&mut analyzer);
1423            analyzer.result
1424        });
1425        cfg
1426    }

1545fn try_add_branch<'e, Exec: ExecutionState<'e>, S: AnalysisState>(
1546    analysis: &mut FuncAnalysis<'e, Exec, S>,
1547    state: (Exec, S),
1548    to: Operand<'e>,
1549) -> Option<Exec::VirtualAddress> {
1550    match to.if_constant() {
1551        Some(s) => {
1552            let binary = analysis.binary;
1553            let code_section = binary.code_section();
1554            let address = Exec::VirtualAddress::from_u64(s);
1555            let rva = Rva(binary.try_rva_32(address)?);
1556            let code_offset = code_section.virtual_address;
1557            let code_len = code_section.data.len() as u32;
1558            let invalid_dest = address < code_offset || address >= code_offset + code_len;
1559            if !invalid_dest {
1560                analysis.add_unchecked_branch(rva, state);
1561            } else {
1562                trace!("Destination {:x} is out of binary bounds", address);
1563                // Add cfg node to keep cfg sensible
1564                // (Adding all branches and checking for binary bounds after states have
1565                // been merged could be better)
1566                analysis.cfg.add_node(rva, CfgNode {
1567                    out_edges: CfgOutEdges::None,
1568                    state: CfgState {
1569                        data: state,
1570                        phantom: Default::default(),
1571                        preceding_branch: PrecedingBranchAddr::multiple(),
1572                    },
1573                    end_address: Rva(rva.0.wrapping_add(1)),
1574                    distance: 0,
1575                });
1576            }
1577            Some(address)
1578        }
1579        None => {
1580            None
1581        }
1582    }
1583}

1593fn update_analysis_for_user_requested_end<'e: 'b, 'b, E, S>(
1594    // Expected to always be AnalysisUpdateResult::Continue
1595    control_ref: &mut AnalysisUpdateResult<'e, 'b, E, S>,
1596    // Option so that it can be moved out of. Expected to always be Some
1597    cfg_node_opt: &mut Option<CfgNode<'e, CfgState<'e, E, S>>>,
1598)
1599where E: ExecutionState<'e> + 'b,
1600      S: AnalysisState,
1601{
1602    let control = match mem::replace(control_ref, AnalysisUpdateResult::End(None)) {
1603        AnalysisUpdateResult::Continue(c) => c,
1604        // Unreachable
1605        AnalysisUpdateResult::End(_) => return,
1606    };
1607    if let Some(mut cfg_node) = cfg_node_opt.take() {
1608        let binary_base_u32 = control.analysis.binary.base().as_u64() as u32;
1609        cfg_node.end_address =
1610            Rva((control.address.as_u64() as u32).wrapping_sub(binary_base_u32));
1611        let branch_start =
1612            Rva((control.branch_start.as_u64() as u32).wrapping_sub(binary_base_u32));
1613        control.analysis.cfg.add_node(branch_start, cfg_node);
1614    }
1615    if control.end == Some(End::ContinueBranch) {
1616        let analysis = control.analysis;
1617        let state = control.state;
1618        analysis.add_unchecked_branch(control.continue_at_address, state);
1619    }
1620}

1622// A separate function to minimize binary size
1623fn update_analysis_for_operation<'e: 'b, 'b, E, S>(
1624    // Expected to always be AnalysisUpdateResult::Continue
1625    // On end gets overwritten to AnalysisUpdateResult::End
1626    control_ref: &mut AnalysisUpdateResult<'e, 'b, E, S>,
1627    op: &Operation<'e>,
1628    // Option so that it can be moved out of. Expected to always be Some
1629    cfg_node_opt: &mut Option<CfgNode<'e, CfgState<'e, E, S>>>,
1630)
1631where E: ExecutionState<'e> + 'b,
1632      S: AnalysisState,
1633{
1634    let control = match control_ref {
1635        AnalysisUpdateResult::Continue(ref mut c) => c,
1636        AnalysisUpdateResult::End(_) => return,
1637    };
1638    match op {
1639        Operation::Jump { .. } | Operation::Return(..) | Operation::Error(..) => {
1640            let cfg_node = match cfg_node_opt {
1641                Some(ref mut s) => s,
1642                None => return,
1643            };
1644            let end = control.end;
1645            let control = match mem::replace(control_ref, AnalysisUpdateResult::End(end)) {
1646                AnalysisUpdateResult::Continue(c) => c,
1647                // Unreachable
1648                AnalysisUpdateResult::End(_) => return,
1649            };
1650            let address = control.address;
1651            if let Operation::Jump { condition, to } = *op {
1652                let state = control.state;
1653                update_analysis_for_jump(
1654                    control.analysis,
1655                    state,
1656                    condition,
1657                    to,
1658                    address,
1659                    u32::from(control.instruction_length),
1660                    &mut cfg_node.out_edges,
1661                );
1662            } else {
1663                drop(control.state);
1664            }
1665            let binary_base_u32 = control.analysis.binary.base().as_u64() as u32;
1666            cfg_node.end_address = Rva((address.as_u64() as u32).wrapping_sub(binary_base_u32));
1667            if let Some(cfg_node) = cfg_node_opt.take() {
1668                let branch_start = Rva(
1669                    (control.branch_start.as_u64() as u32).wrapping_sub(binary_base_u32)
1670                );
1671                control.analysis.cfg.add_node(branch_start, cfg_node);
1672            }
1673        }
1674        o => {
1675            control.state.0.update(o);
1676        }
1677    }
1678}

1680fn update_analysis_for_jump<'e, Exec: ExecutionState<'e>, S: AnalysisState>(
1681    analysis: &mut FuncAnalysis<'e, Exec, S>,
1682    mut state: (Exec, S),
1683    condition: Operand<'e>,
1684    to: Operand<'e>,
1685    address: Exec::VirtualAddress,
1686    instruction_len: u32,
1687    cfg_out_edge: &mut CfgOutEdges<'e>,
1688) {
1689    /// Returns address of the table,
1690    /// operand that is being used to index the table,
1691    /// constant that is being added to any values read from the table,
1692    /// and size in bytes of one value in table.
1693    /// E.g. dest = ret.2 + read_'ret.3'_bytes(ret.0 + ret.1 * ret.3)
1694    fn is_switch_jump<'e, Va: exec_state::VirtualAddress>(
1695        to: Operand<'e>,
1696        ctx: OperandCtx<'e>,
1697    ) -> Option<(Va, Operand<'e>, u64, MemAccessSize)> {
1698        // remove (x + y) & ffff_ffff mask on 32bit
1699        let to = match to.if_and_with_const() {
1700            Some((inner, c)) => {
1701                if c == Va::max_value().as_u64() {
1702                    inner
1703                } else {
1704                    to
1705                }
1706            }
1707            None => to,
1708        };
1709        let (base, mem) = match to.if_arithmetic_add() {
1710            Some((l, r)) => (r.if_constant()?, l),
1711            None => (0, to),
1712        };
1713        mem.if_memory()
1714            .and_then(|mem| {
1715                let (mut index, mut switch_table) = mem.address();
1716                let mut and_mask = None;
1717                if let Some((l, r)) = index.if_arithmetic_and() {
1718                    // Index such as (((rcx * 4) - 34) & 3fffffffc) for Mem32
1719                    // should be understood as index = rcx, table offset -= 0xd
1720                    let mask = r.if_constant()?;
1721                    if mask as u32 & mem.size.bytes() - 1 != 0 {
1722                        return None;
1723                    }
1724                    let (inner, offset) = l.if_arithmetic_sub()
1725                        .and_then(|(l, r)| {
1726                            let c = r.if_constant()?;
1727                            Some((l, 0u64.wrapping_sub(c)))
1728                        })
1729                        .or_else(|| {
1730                            let (l, r) = l.if_arithmetic_add()?;
1731                            let c = r.if_constant()?;
1732                            Some((l, c))
1733                        })?;
1734                    index = inner;
1735                    and_mask = Some(mask);
1736                    switch_table = switch_table.wrapping_add(offset);
1737                }
1738                let (index, r) = index.if_arithmetic_mul()?;
1739                let c = u32::try_from(r.if_constant()?).ok()?;
1740                if c == Va::SIZE || base != 0 {
1741                    let (mem_size, shift) = match c {
1742                        1 => (MemAccessSize::Mem8, 0),
1743                        2 => (MemAccessSize::Mem16, 1),
1744                        4 => (MemAccessSize::Mem32, 2),
1745                        8 => (MemAccessSize::Mem64, 3),
1746                        _ => return None,
1747                    };
1748                    let index = match and_mask {
1749                        Some(s) => ctx.and_const(index, s >> shift),
1750                        None => index,
1751                    };
1752                    Some((Va::from_u64(switch_table), index, base, mem_size))
1753                } else {
1754                    None
1755                }
1756            })
1757    }

1759    fn switch_cases<'e, Va: VaTrait>(
1760        binary: &'e BinaryFile<Va>,
1761        mem_size: MemAccessSize,
1762        switch_table_addr: Va,
1763        limits: (u64, u64),
1764        base_addr: u64,
1765    ) -> Option<impl Iterator<Item = Va> + 'e> {
1766        let size_shift = match mem_size {
1767            MemAccessSize::Mem8 => 0u8,
1768            MemAccessSize::Mem16 => 1,
1769            MemAccessSize::Mem32 => 2,
1770            MemAccessSize::Mem64 => 3,
1771        };
1772        let case_size = 1u32 << size_shift;
1773        let size_mask = mem_size.mask();
1774
1775        let code_section = binary.code_section();
1776        let code_offset = code_section.virtual_address;
1777        let code_len = code_section.data.len() as u32;
1778        let code_end = code_offset + code_len;
1779
1780        // Truncating to u32.. Probably fine since BinaryFile is not expected to be over 4GB..
1781        // Will have to sign extend later on in the function in order to handle negative offsets
1782        // correctly.
1783        let start = limits.0 as u32;
1784        let end = limits.1 as u32;
1785        let va_mask = Va::max_value().as_u64() as u32;
1786        let mut pos = (start << size_shift) & va_mask;
1787        let end = (end << size_shift) & va_mask;
1788
1789        // Handle reads of switch table values as `(u64(data - shift) >> shift) & mask`,
1790        // shift is 0 if switch table is right at section start to make `data - shift` not
1791        // overflow past start, otherwise it is set so that `u64(data - shift)` won't have
1792        // to read past section end in case switch table goes that far.
1793        let switch_table_shift;
1794        let mut switch_table_data;
1795        let switch_table_section = binary.section_by_addr(switch_table_addr)?;
1796        let section_relative = (switch_table_addr.as_u64() as usize)
1797            .wrapping_sub(switch_table_section.virtual_address.as_u64() as usize)
1798            .wrapping_add(pos as i32 as isize as usize);
1799        if section_relative < 8 {
1800            switch_table_shift = 0u32;
1801            switch_table_data = switch_table_section.data.get(section_relative..)?;
1802        } else {
1803            let shift_bytes = 8u32.wrapping_sub(case_size);
1804            switch_table_shift = shift_bytes << 3;
1805            switch_table_data = switch_table_section.data
1806                .get((section_relative - shift_bytes as usize)..)?;
1807        }
1808
1809        let mut reached_end = false;
1810        Some(std::iter::from_fn(move || {
1811            if reached_end {
1812                return None;
1813            }
1814
1815            if !binary.relocs.is_empty() {
1816                let addr = switch_table_addr + pos;
1817                if binary.relocs.binary_search(&addr).is_err() {
1818                    return None;
1819                }
1820            }
1821            reached_end = pos == end;
1822            pos = pos.wrapping_add(case_size) & va_mask;
1823            if switch_table_data.len() < 8 {
1824                return None;
1825            }
1826            let switch_table_value =
1827                (LittleEndian::read_u64(&switch_table_data) >> switch_table_shift) & size_mask;
1828            switch_table_data = switch_table_data.get((case_size as usize)..)?;
1829            let jump_addr = Va::from_u64(base_addr.wrapping_add(switch_table_value));
1830            if jump_addr >= code_offset && jump_addr < code_end {
1831                Some(jump_addr)
1832            } else {
1833                None
1834            }
1835        }))
1836    }
1837
1838    state.0.maybe_convert_memory_immutable(16);
1839    let condition_resolved = if condition.needs_resolve() {
1840        state.0.resolve_apply_constraints(condition)
1841    } else {
1842        condition
1843    };
1844
1845    let ctx = analysis.operand_ctx;
1846    if condition_resolved == ctx.const_0() {
1847        // Never jump
1848        let address = address + instruction_len;
1849        if condition != ctx.const_0() {
1850            let constraint = analysis.operand_ctx.eq_const(condition, 0);
1851            state.0.add_unresolved_constraint(Constraint::new(constraint));
1852        }
1853        if let Some(rva) = analysis.binary.try_rva_32(address) {
1854            *cfg_out_edge = CfgOutEdges::Single(NodeLink::new(Rva(rva)));
1855            analysis.add_unchecked_branch(Rva(rva), state);
1856        }
1857    } else if condition_resolved == ctx.const_1() {
1858        // Always jump
1859        let to = state.0.resolve(to);
1860        let is_switch = is_switch_jump::<Exec::VirtualAddress>(to, ctx);
1861        if let Some((switch_table_addr, index, base_addr, mem_size)) = is_switch {
1862            let binary = analysis.binary;
1863            let limits = state.0.value_limits(index);
1864            let case_guess = (limits.1.wrapping_sub(limits.0) as usize)
1865                .wrapping_add(1)
1866                .min(128);
1867            let mut cases = Vec::with_capacity(case_guess);
1868
1869            let mut case_iter =
1870                switch_cases(binary, mem_size, switch_table_addr, limits, base_addr);
1871            if let Some(ref mut case_iter) = case_iter {
1872                for addr in case_iter {
1873                    if let Some(rva) = binary.try_rva_32(addr) {
1874                        analysis.add_unchecked_branch(Rva(rva), state.clone());
1875                        cases.push(NodeLink::new(Rva(rva)));
1876                    }
1877                }
1878
1879                if !cases.is_empty() {
1880                    *cfg_out_edge = CfgOutEdges::Switch(cases, index);
1881                }
1882            }
1883        } else {
1884            if condition != ctx.const_1() {
1885                state.0.add_unresolved_constraint(Constraint::new(condition));
1886            }
1887            let dest = try_add_branch(analysis, state, to);
1888            *cfg_out_edge = CfgOutEdges::Single(
1889                dest.and_then(|dest| {
1890                    Some(NodeLink::new(Rva(analysis.binary.try_rva_32(dest)?)))
1891                }).unwrap_or_else(NodeLink::unknown)
1892            );
1893        }
1894    } else {
1895        let binary = analysis.binary;
1896        let no_jump_addr = address + instruction_len;
1897        if let Some(rva) = binary.try_rva_32(no_jump_addr) {
1898            let to = state.0.resolve(to);
1899            let (jump, no_jump) = state.0.assume_jump_flag(condition, condition_resolved);
1900            let jump_state = (jump, state.1.clone());
1901            let no_jump_state = (no_jump, state.1);
1902            analysis.add_unchecked_branch(Rva(rva), no_jump_state);
1903            let dest = try_add_branch(analysis, jump_state, to);
1904            *cfg_out_edge = CfgOutEdges::Branch(
1905                NodeLink::new(Rva(rva)),
1906                OutEdgeCondition {
1907                    node: dest.and_then(|dest| {
1908                        Some(NodeLink::new(Rva(binary.try_rva_32(dest)?)))
1909                    }).unwrap_or_else(NodeLink::unknown),
1910                    condition,
1911                },
1912            );
1913        }
1914    }
1915}
